🗂 URLs de Google Sheets por Año

Cada año se conecta a una hoja de cálculo diferente. Configura la URL de cada año y usa los botones para sincronizar solo ese año.

⚡ Año Activo: —

Las operaciones de abajo afectan solo al año activo y su hoja configurada.

🟢 ¿Cómo configurar Google Sheets por año?

1
Abre tu Google Sheet (crea uno por año si no tienes)
2
Ve a Extensiones → Apps Script — borra todo el código existente y pega el código de abajo
3
Guarda con Ctrl+S, luego clic en "Implementar" → "Nueva implementación"
4
Tipo: Aplicación web · Ejecutar como: Tu cuenta · Acceso: Cualquier persona → clic Implementar
5
Google te pedirá autorizar permisos — acepta todo. Copia la URL de implementación que aparece
6
Pega esa URL en el campo del año correspondiente arriba y guarda. ¡Listo! Ya puedes exportar e importar.
!
Importante: Cada vez que edites el código debes hacer una nueva implementación (no editar la existente). Si ves error CORS o "Failed to fetch", re-implementa el script.
// ── ROMAN Importaciones – Google Apps Script v4 (Colaborativo) ── // INSTRUCCIONES: // 1. Abre tu Google Sheet → Extensiones → Apps Script // 2. Borra TODO el código existente → pega este código → Guarda (Ctrl+S) // 3. Clic en "Implementar" → "Nueva implementación" // 4. Tipo: Aplicación web | Ejecutar como: TU CUENTA | Acceso: Cualquier persona // 5. Autoriza permisos → copia la URL /exec → pégala en la app // ⚠️ Cada vez que modifiques el código: NUEVA implementación (no editar existente) var DATA_SHEET = ''; // Vacío = hoja activa (datos principales) var PRESENCE_SHEET = '_presencia'; // Hoja oculta de usuarios activos function getDataSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); return DATA_SHEET ? (ss.getSheetByName(DATA_SHEET) || ss.getActiveSheet()) : ss.getActiveSheet(); } function getPresenceSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(PRESENCE_SHEET); if (!sh) { sh = ss.insertSheet(PRESENCE_SHEET); sh.hideSheet(); sh.getRange(1,1,1,4).setValues([['userId','nombre','lastSeen','year']]); } return sh; } function makeOutput(obj) { var out = ContentService.createTextOutput(JSON.stringify(obj)); out.setMimeType(ContentService.MimeType.JSON); return out; } // ── Limpiar presencias viejas (más de 2 minutos) ── function cleanPresence(sheet) { var data = sheet.getDataRange().getValues(); var now = new Date().getTime(); var toDelete = []; for (var i = data.length - 1; i >= 1; i--) { var ts = new Date(data[i][2]).getTime(); if (isNaN(ts) || (now - ts) > 120000) toDelete.push(i + 1); } toDelete.forEach(function(r) { sheet.deleteRow(r); }); } // ── Registrar presencia de un usuario ── function upsertPresence(sheet, userId, nombre, year) { var data = sheet.getDataRange().getValues(); var now = new Date().toISOString(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]) === String(userId)) { sheet.getRange(i+1, 1, 1, 4).setValues([[userId, nombre, now, year]]); return; } } sheet.appendRow([userId, nombre, now, year]); } // ── Leer usuarios activos ── function getActiveUsers(sheet) { cleanPresence(sheet); var data = sheet.getDataRange().getValues(); var users = []; for (var i = 1; i < data.length; i++) { if (data[i][0]) users.push({id: data[i][0], nombre: String(data[i][1]), year: String(data[i][3])}); } return users; } function doGet(e) { try { var sheet = getDataSheet(); var data = sheet.getDataRange().getValues(); // Registrar presencia si viene userId var activeUsers = []; try { var ps = getPresenceSheet(); if (e.parameter && e.parameter.userId) { upsertPresence(ps, e.parameter.userId, e.parameter.nombre || 'Usuario', e.parameter.year || ''); } activeUsers = getActiveUsers(ps); } catch(pe) {} return makeOutput({ok: true, data: data, sheet: sheet.getName(), rows: data.length, activeUsers: activeUsers}); } catch(err) { return makeOutput({ok: false, msg: String(err.message)}); } } function doPost(e) { try { var payload = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = getDataSheet(); // Registrar presencia en toda operación POST var activeUsers = []; try { var ps = getPresenceSheet(); if (payload.userId) { upsertPresence(ps, payload.userId, payload.userName || 'Usuario', payload.year || ''); } activeUsers = getActiveUsers(ps); } catch(pe) {} // ── EXPORTAR ── if (payload.action === 'export') { sheet.clearContents(); if (payload.rows && payload.rows.length > 0) { sheet.getRange(1,1,payload.rows.length,payload.rows[0].length).setValues(payload.rows); sheet.getRange(1,1,1,payload.rows[0].length).setBackground('#1F3864').setFontColor('#FFFFFF').setFontWeight('bold'); sheet.setFrozenRows(1); } return makeOutput({ok: true, msg: 'Exportados: '+(payload.rows.length-1)+' registros', activeUsers: activeUsers}); } // ── SINCRONIZAR (exporta local + devuelve datos actuales del Sheet) ── if (payload.action === 'sync') { // Guardar datos locales if (payload.rows && payload.rows.length > 0) { sheet.clearContents(); sheet.getRange(1,1,payload.rows.length,payload.rows[0].length).setValues(payload.rows); sheet.getRange(1,1,1,payload.rows[0].length).setBackground('#1F3864').setFontColor('#FFFFFF').setFontWeight('bold'); sheet.setFrozenRows(1); } // Devolver datos frescos (ya incluye lo que se guardó) var fresh = sheet.getDataRange().getValues(); return makeOutput({ok: true, data: fresh, msg: 'Sync OK: '+(payload.rows.length-1)+' registros', activeUsers: activeUsers}); } // ── RESPALDO ── if (payload.action === 'backup') { var ts = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyyMMdd_HHmm'); var backupName = 'BCK_'+(payload.year||'')+'_'+ts; var old = ss.getSheetByName(backupName); if (old) ss.deleteSheet(old); var bSheet = sheet.copyTo(ss); bSheet.setName(backupName); bSheet.setTabColor('#FF6D00'); bSheet.hideSheet(); return makeOutput({ok: true, msg: 'Respaldo creado: '+backupName, sheet: backupName, activeUsers: activeUsers}); } // ── SOLO PRESENCIA (ping) ── if (payload.action === 'ping') { return makeOutput({ok: true, activeUsers: activeUsers}); } return makeOutput({ok: false, msg: 'Accion desconocida'}); } catch(err) { return makeOutput({ok: false, msg: String(err.message)}); } }

⚠️ Notas importantes

  • Cada año tiene su propia URL — no mezcles URLs entre años.
  • La exportación sobrescribe los datos en la hoja del año. Haz backup antes.
  • Al importar, se reemplaza la base de datos local del año activo.
  • Si ves error CORS, asegúrate de desplegar como "Cualquier usuario".

📦 ¿Cómo conectar el Catálogo de Productos?

💡
¿Ya tienes el script de Importaciones funcionando?
Solo actualiza el código con el de abajo, haz una nueva implementación y copia la nueva URL.
Puedes usar la misma hoja de Google Sheets — el script crea automáticamente las pestañas Productos e HistorialPrecios.
1
Abre tu Google Sheet → ve a Extensiones → Apps Script
Si no tienes una hoja aún, crea una nueva en sheets.new
2
Selecciona todo el código existente (Ctrl+A) y bórralo. Luego pega el código del bloque de abajo y guarda con Ctrl+S
Usa el botón verde "Copiar código" para copiar fácilmente
3
Haz clic en "Implementar" (botón azul arriba a la derecha) → "Nueva implementación"
⚠️ Siempre "Nueva implementación", nunca "Editar implementación existente"
4
Configura así:
Tipo:Aplicación web
Ejecutar como:Tu cuenta de Google
Acceso:Cualquier persona (Anyone)
5
Haz clic en "Implementar" → Google te pedirá autorizar permisos — acepta todo
Si aparece "Google no ha verificado la app" → clic en "Configuración avanzada" → "Ir a (nombre del proyecto)"
6
Copia la URL que aparece (termina en /exec) y pégala en el campo "URL del Apps Script — Productos" del panel verde de arriba → clic "Guardar URL"
7
Haz clic en "🔌 Probar conexión" — si aparece ✅ Conexión OK ya está todo listo
Luego puedes usar "📤 Exportar Productos → Sheets" para subir tu catálogo
❌
Si ves error "Failed to fetch" o CORS: volviste a editar la implementación existente. Debes hacer siempre Nueva implementación y usar la nueva URL generada.
// ── ROMAN Importaciones + Productos – Google Apps Script v5 ── // Incluye: Importaciones por año + Catálogo de Productos + Historial de Precios // INSTRUCCIONES: // 1. Abre tu Google Sheet → Extensiones → Apps Script // 2. Borra TODO el código existente → pega este código → Guarda (Ctrl+S) // 3. Clic en "Implementar" → "Nueva implementación" // 4. Tipo: Aplicación web | Ejecutar como: TU CUENTA | Acceso: Cualquier persona // 5. Autoriza permisos → copia la URL /exec → pégala en la app // ⚠️ Cada vez que modifiques el código: NUEVA implementación (no editar existente) var DATA_SHEET = ''; // Vacío = hoja activa var PRESENCE_SHEET = '_presencia'; // Hoja oculta de usuarios activos var PROD_SHEET = 'Productos'; // Hoja para el catálogo de productos var HIST_SHEET = 'HistorialPrecios'; // Hoja para historial de precios function getDataSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); return DATA_SHEET ? (ss.getSheetByName(DATA_SHEET) || ss.getActiveSheet()) : ss.getActiveSheet(); } function getOrCreateSheet(name, headers) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); if (headers && headers.length) { sh.getRange(1,1,1,headers.length).setValues([headers]) .setBackground('#1F3864').setFontColor('#FFFFFF').setFontWeight('bold'); sh.setFrozenRows(1); } } return sh; } function getPresenceSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(PRESENCE_SHEET); if (!sh) { sh = ss.insertSheet(PRESENCE_SHEET); sh.hideSheet(); sh.getRange(1,1,1,4).setValues([['userId','nombre','lastSeen','year']]); } return sh; } function makeOutput(obj) { var out = ContentService.createTextOutput(JSON.stringify(obj)); out.setMimeType(ContentService.MimeType.JSON); return out; } function cleanPresence(sheet) { var data = sheet.getDataRange().getValues(); var now = new Date().getTime(); var toDelete = []; for (var i = data.length - 1; i >= 1; i--) { var ts = new Date(data[i][2]).getTime(); if (isNaN(ts) || (now - ts) > 120000) toDelete.push(i + 1); } toDelete.forEach(function(r) { sheet.deleteRow(r); }); } function upsertPresence(sheet, userId, nombre, year) { var data = sheet.getDataRange().getValues(); var now = new Date().toISOString(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]) === String(userId)) { sheet.getRange(i+1,1,1,4).setValues([[userId,nombre,now,year]]); return; } } sheet.appendRow([userId,nombre,now,year]); } function getActiveUsers(sheet) { cleanPresence(sheet); var data = sheet.getDataRange().getValues(); var users = []; for (var i = 1; i < data.length; i++) { if (data[i][0]) users.push({id:data[i][0],nombre:String(data[i][1]),year:String(data[i][3])}); } return users; } function doGet(e) { try { var action = e.parameter && e.parameter.action ? e.parameter.action : ''; // ── GET Productos ── if (action === 'getProductos') { var ps = getOrCreateSheet(PROD_SHEET, ['id','cod','nombre','desc','cat','prov1','precio1','costo1','fechaPrecio1','fechaCosto1','prov2','precio2','costo2','fechaPrecio2','fechaCosto2','prov3','precio3','costo3','fechaPrecio3','fechaCosto3','createdAt','updatedAt']); var hs = getOrCreateSheet(HIST_SHEET, ['id','prodId','proveedor','precio','precioAnterior','fecha']); var pd = ps.getDataRange().getValues(); var hd = hs.getDataRange().getValues(); var productos = []; for (var i = 1; i < pd.length; i++) { // Saltar solo si la fila está completamente vacía (sin id, sin cod, sin nombre) if (!pd[i][0] && !pd[i][1] && !pd[i][2]) continue; try { var proveedores = []; for (var k=0; k<3; k++) { var base = 5 + k*5; // prov,precio,costo,fechaPrecio,fechaCosto var pvNombre = pd[i][base]; if (pvNombre) { proveedores.push({ nombre: String(pvNombre), precio: pd[i][base+1] || '', costo: pd[i][base+2] || '', fechaPrecio: pd[i][base+3] ? String(pd[i][base+3]) : '', fechaCosto: pd[i][base+4] ? String(pd[i][base+4]) : '' }); } } var codVal = pd[i][1] ? String(pd[i][1]).trim() : ''; var nombreVal = String(pd[i][2]||'').replace(/""/g,'"'); var descVal = String(pd[i][3]||'').replace(/""/g,'"'); var idVal = pd[i][0] ? String(pd[i][0]) : (codVal ? ('p_cod_'+codVal) : ('p_row_'+i+'_'+new Date().getTime())); productos.push({ id: idVal, cod: codVal, nombre: nombreVal, desc: descVal, cat: pd[i][4], proveedores: proveedores, createdAt: pd[i][20]||'', updatedAt: pd[i][21]||'' }); } catch(pe) {} } var historial = []; for (var j = 1; j < hd.length; j++) { if (!hd[j][0]) continue; historial.push({ id: hd[j][0], prodId: hd[j][1], proveedor: hd[j][2], precio: hd[j][3], precioAnterior: hd[j][4], fecha: hd[j][5] }); } return makeOutput({ok:true, productos:productos, historial:historial}); } // ── Ping ── if (action === 'ping') { return makeOutput({ok:true, msg:'pong'}); } // ── GET Importaciones (default) ── var sheet = getDataSheet(); var data = sheet.getDataRange().getValues(); var activeUsers = []; try { var prs = getPresenceSheet(); if (e.parameter && e.parameter.userId) { upsertPresence(prs, e.parameter.userId, e.parameter.nombre||'Usuario', e.parameter.year||''); } activeUsers = getActiveUsers(prs); } catch(pe) {} return makeOutput({ok:true, data:data, sheet:sheet.getName(), rows:data.length, activeUsers:activeUsers}); } catch(err) { return makeOutput({ok:false, msg:String(err.message)}); } } function doPost(e) { try { var payload = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var activeUsers = []; try { var ps = getPresenceSheet(); if (payload.userId) upsertPresence(ps, payload.userId, payload.userName||'Usuario', payload.year||''); activeUsers = getActiveUsers(ps); } catch(pe) {} // ── SYNC PRODUCTOS ── if (payload.action === 'syncProductos') { var prods = payload.productos || []; var hist = payload.historial || []; var prodSheet = getOrCreateSheet(PROD_SHEET, ['id','cod','nombre','desc','cat','prov1','precio1','costo1','fechaPrecio1','fechaCosto1','prov2','precio2','costo2','fechaPrecio2','fechaCosto2','prov3','precio3','costo3','fechaPrecio3','fechaCosto3','createdAt','updatedAt']); prodSheet.clearContents(); prodSheet.getRange(1,1,1,22).setValues([['id','cod','nombre','desc','cat','prov1','precio1','costo1','fechaPrecio1','fechaCosto1','prov2','precio2','costo2','fechaPrecio2','fechaCosto2','prov3','precio3','costo3','fechaPrecio3','fechaCosto3','createdAt','updatedAt']]) .setBackground('#16693A').setFontColor('#FFFFFF').setFontWeight('bold'); }); prodSheet.getRange(2,1,prodRows.length,22).setValues(prodRows); } var histSheet = getOrCreateSheet(HIST_SHEET, ['id','prodId','proveedor','precio','precioAnterior','fecha']); histSheet.clearContents(); histSheet.getRange(1,1,1,6).setValues([['id','prodId','proveedor','precio','precioAnterior','fecha']]) .setBackground('#1F4E79').setFontColor('#FFFFFF').setFontWeight('bold'); if (hist.length) { var histRows = hist.map(function(h){ return [h.id||'', h.prodId||'', h.proveedor||'', h.precio||'', h.precioAnterior||'', h.fecha||'']; }); histSheet.getRange(2,1,histRows.length,6).setValues(histRows); } return makeOutput({ok:true, msg:'Sync OK: '+prods.length+' productos, '+hist.length+' historial', activeUsers:activeUsers}); } var sheet = getDataSheet(); // ── EXPORTAR ── if (payload.action === 'export') { sheet.clearContents(); if (payload.rows && payload.rows.length > 0) { sheet.getRange(1,1,payload.rows.length,payload.rows[0].length).setValues(payload.rows); sheet.getRange(1,1,1,payload.rows[0].length).setBackground('#1F3864').setFontColor('#FFFFFF').setFontWeight('bold'); sheet.setFrozenRows(1); } return makeOutput({ok:true, msg:'Exportados: '+(payload.rows.length-1)+' registros', activeUsers:activeUsers}); } // ── SINCRONIZAR ── if (payload.action === 'sync') { if (payload.rows && payload.rows.length > 0) { sheet.clearContents(); sheet.getRange(1,1,payload.rows.length,payload.rows[0].length).setValues(payload.rows); sheet.getRange(1,1,1,payload.rows[0].length).setBackground('#1F3864').setFontColor('#FFFFFF').setFontWeight('bold'); sheet.setFrozenRows(1); } var fresh = sheet.getDataRange().getValues(); return makeOutput({ok:true, data:fresh, msg:'Sync OK: '+(payload.rows.length-1)+' registros', activeUsers:activeUsers});
